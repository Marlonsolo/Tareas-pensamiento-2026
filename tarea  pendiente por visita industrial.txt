cantidad = int(input("¿Cuántos números va a ingresar?: "))

lista_numeros = []

for i in range(cantidad):
    n = int(input(f"Ingrese el valor {i+1}: "))
    lista_numeros.append(n)

pares = 0
impares = 0
positivos = 0
negativos = 0
ceros = 0
suma_total = 0

mayor = lista_numeros[0]
menor = lista_numeros[0]

for num in lista_numeros:
    suma_total += num
    
    if num % 2 == 0:
        pares += 1
    else:
        impares += 1
        
    if num > 0:
        positivos += 1
    elif num < 0:
        negativos += 1
    else:
        ceros += 1
        
    if num > mayor:
        mayor = num
    if num < menor:
        menor = num

print("\n--- RESULTADOS ---")
print("Lista completa:", lista_numeros)
print("Pares:", pares)
print("Impares:", impares)
print("Positivos:", positivos)
print("Negativos:", negativos)
print("Ceros:", ceros)
print("Suma total:", suma_total)
print("Número mayor:", mayor)
print("Número menor:", menor)
