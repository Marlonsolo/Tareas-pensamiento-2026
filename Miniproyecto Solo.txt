import tkinter as tk
from tkinter import ttk, messagebox

dias = ["Lunes", "Martes", "Miércoles", "Jueves", "Viernes"]
niveles_azucar = [130, 160, 95, 175, 160]
niveles_sal = [2000, 2400, 1800, 2400, 2700]
presion = [115, 130, 110, 125, 175]

root = tk.Tk()
root.title("Evaluación de Salud - Laboratorio")
root.geometry("400x400")

label_instruccion = tk.Label(root, text="Seleccione el día para ver los resultados:")
label_instruccion.pack(pady=10)

combo_dias = ttk.Combobox(root, values=dias, state="readonly")
combo_dias.pack(pady=5)

texto_resultados = tk.Text(root, height=12, width=45)
texto_resultados.pack(pady=10)

def mostrar_datos():
    seleccion = combo_dias.get()
    if not seleccion:
        messagebox.showwarning("Advertencia", "Por favor seleccione un día")
        return
    
    indice = dias.index(seleccion)
    azucar = niveles_azucar[indice]
    sal = niveles_sal[indice]
    sistolica = presion[indice]
    
    texto_resultados.delete("1.0", tk.END)
    texto_resultados.insert(tk.END, f"RESULTADOS: {seleccion}\n")
    texto_resultados.insert(tk.END, "-"*30 + "\n")
    
    if 70 <= azucar <= 140:
        texto_resultados.insert(tk.END, f"Azúcar: {azucar} mg/dL - Normal\n")
    else:
        texto_resultados.insert(tk.END, f"ALERTA: Azúcar {azucar} mg/dL fuera de rango\n")
        
    if sal <= 2300:
        texto_resultados.insert(tk.END, f"Sal: {sal} mg - Normal\n")
    else:
        texto_resultados.insert(tk.END, f"ALERTA: Consumo de sal {sal} mg excedido\n")
        
    if sistolica < 120:
        texto_resultados.insert(tk.END, f"Presión: {sistolica} mmHg - Normal\n")
    elif 120 <= sistolica <= 129:
        texto_resultados.insert(tk.END, f"Presión: {sistolica} mmHg - Elevada\n")
    elif 130 <= sistolica <= 139:
        texto_resultados.insert(tk.END, f"Presión: {sistolica} mmHg - Hipertensión Etapa 1\n")
    else:
        texto_resultados.insert(tk.END, f"ALERTA: Presión {sistolica} mmHg - Etapa 2 (Intervención)\n")

    suma_azucar = sum(niveles_azucar)
    suma_sal = sum(niveles_sal)
    suma_presion = sum(presion)
    prom_azucar = suma_azucar / len(dias)
    prom_sal = suma_sal / len(dias)
    prom_presion = suma_presion / len(dias)

    texto_resultados.insert(tk.END, "\nPROMEDIOS SEMANALES:\n")
    texto_resultados.insert(tk.END, f"Azúcar: {prom_azucar:.1f} | Sal: {prom_sal:.1f} | Presión: {prom_presion:.1f}")

boton_consultar = tk.Button(root, text="Consultar Día", command=mostrar_datos)
boton_consultar.pack(pady=5)

root.mainloop()
